#include <Adafruit_GFX.h>
#include <Adafruit_ST7735.h>
#include <SPI.h>

// SPI pins (explicit)
#define TFT_SCK   18
#define TFT_MOSI  23
// MISO isn't used by ST7735, but SPI.begin still wants the slot:
#define TFT_MISO  -1

#define TFT_CS   16
#define TFT_DC   22
#define TFT_RST 21

const int w = 2;

Adafruit_ST7735 tft = Adafruit_ST7735(TFT_CS, TFT_DC, TFT_MOSI, TFT_SCK, TFT_RST);

void setup() {
  //SPI.begin(TFT_SCK, TFT_MISO, TFT_MOSI);
  tft.initR(INITR_BLACKTAB);
  tft.setRotation(0);
  
  pinMode(w, OUTPUT);
}

void loop() {
  tft.fillScreen(ST77XX_RED);
  digitalWrite(w, HIGH);
  delay(1000);

  tft.fillScreen(ST77XX_GREEN);
  digitalWrite(w, LOW);
  delay(1000);

  tft.fillScreen(ST77XX_BLUE);
  digitalWrite(w, HIGH);
  delay(1000);

  tft.fillScreen(ST77XX_WHITE);
  digitalWrite(w, LOW);
  delay(1000);
}
